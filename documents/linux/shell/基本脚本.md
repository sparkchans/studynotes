```shell
# 在命令行中使用多个命令
date; who

# 显示日期
#/bin/bash
# 若要显示在一行使用 echo -n 参数
echo "Current date is:"
date

# 使用set命令显示当前环境山所有的环境变量，使用环境变量在前面加上$符号
$JAVA_HOME

# 将命令执行的结果赋值给环境变量
currentDate=$(date)

# 重定向符号，> 表示全新增加，>> 表示在结尾处增加，<< 表示内联输入重定向，需要指定输入输入的开始和结束
wc << EOF
test1
test2
test3
EOF

# Linux 提供一个专门的变量 $? 来保存上个已执行命令的退出状态码。对于需要进行检查的命令，必须在其运行完毕后立刻查看或使用 $? 变量。0 表示正常退出，126 表示没有权限，130 表示 ctl+c 终止的命令。可以通过 exit 命令来自定义退出码。
exit 170

```
